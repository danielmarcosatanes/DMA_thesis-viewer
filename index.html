    <p>Select a molecule to view:</p>

    

    <!-- Dropdown menu to select the molecule -->

    <select id="moleculeSelector" onchange="loadMolecule()">

        <option value="molecules/molecule1.xyz">Molecule 1</option>

        <option value="molecules/molecule2.xyz">Molecule 2</option>

        <!-- Add more options here for additional molecules -->

    </select>

    

    <!-- The container for the JSmol viewer -->

    <div id="jsmolContainer" style="width: 600px; height: 400px; border: 1px solid black;"></div>


    <script type="text/javascript">

        var Info = {

            width: 600,

            height: 400,

            j2sPath: "j2s", // Path to JSmol's j2s folder

            color: "white",  // Background color

            use: "HTML5",    // Use HTML5 for rendering

            script: "load molecules/molecule1.xyz", // Default molecule to load

            disableJ2SLoadMonitor: true,  // Disable loading monitor

            disableInitialConsole: true  // Disable initial console

        };

        Jmol.setDocument(0); // Set the document context for JSmol

        Jmol.getApplet("jsmolApplet", Info); // Initialize the JSmol applet

        document.getElementById("jsmolContainer").innerHTML = Jmol.getAppletHtml("jsmolApplet");


        // Function to load the selected molecule

        function loadMolecule() {

            var moleculePath = document.getElementById('moleculeSelector').value;

            Jmol.script(jsmolApplet, `load ${moleculePath}`);

        }

    </script>
